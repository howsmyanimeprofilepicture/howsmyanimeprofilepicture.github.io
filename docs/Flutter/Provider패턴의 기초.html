class CountProvider extends ChangeNotifier {
  int _count = 0;
​
  int get count {
    return _count;
  }
​
  void add() {
    _count++;
    notifyListeners();
  }
​
  void remove() {
    _count--;
    notifyListeners();
  }
}
​
  1. 우선은 App State를 관리할 모듈(프로바이더)을 분리합시다.
  2. 위 예제에서 해당 모듈의 이름을 CountProvider라고 하겠습니다. 해당 모듈은 ChangeNotifier를 상속받습니다.
  3. 위의 예제에서는 _count라는 state를 선언했습니다. (앞에 언더바가 붙으면 private한 프로퍼티가 됩니다.)
  4. 그리고 getter를 통해 CounterProvider 외부에서 해당 state를 관리하지 못하게했습니다.
  5. 상태값 외에도 상태관리 메서드를 선언합니다. add와 remove를 선언했습니다.
  6. 그리고 notifyListeners()를 호출해서 상태의 변화가 프로바이더를 구독하는 리스너위젯들에게 전달될 수 있게 합니다.
  7. notifyListeners()가 호출되면 리스너위젯들은 (별도로 listen : false를 설정하지 않는 한) 최신 상태값으로 리빌드됩니다.

class MyApp extends StatelessWidget {
    //provider 패턴에 익숙해지면 지역적이고 아주 사소한 상태 관리가 아닌 이상
    //StatefulWidget을 사용할 이유가 없어집니다.
    //StatelessWidget은 위젯 자체의 State는 없습니다.
    //단 provider를 통해 App State에 의해
    // 그리고 필요한 상황에 따라 리빌드가 되는 것입니다.
  CountProvider _countProvider; //CounterProvider 타입으로 변수 하나를 선언합니다.
  @override
  Widget build(BuildContext myAppCtx) {
      // 컨텍스트 구분을 편히 하기 위해 MyApp의 컨텍스트는 myAppCtx라고 했습니다.
    return MaterialApp(
      title: 'Flutter Demo',
      theme: ThemeData(
        primarySwatch: Colors.blue,
      ),
      home: ChangeNotifierProvider(
        create: (BuildContext _) {
            // 아직 확실한 건 아니지만 ChangeNotifierProvider 위젯의 
            // create 콜백의 BuildContext인자는 
            // 특별한 경우가 아닌 한은 따로 참조할 일이 없을 거 같아 언더바로 표시했습니다. 
          return CountProvider();
        },
        child: Builder(
          builder: (builderCtx) { 
              //Builder위젯의 빌드 함수의 context는 builderCtx라고 명명했습니다.
            _countProvider = Provider.of<CountProvider>(
              builderCtx,
              listen: true, 
               //listen 파라미터를 false로 하면 구독하는 프로바이더의 state가 변해도 해당 위젯이 리빌드 되지 않습니다.
            ); //Provider.of<참조하고자하는프로바이더이름>(컨텍스트, listen: true or false)를 통해
              //프로바이더로 접근할 수 있습니다.
              // 만약 컨텍스트를 builderCtx가 아닌 myAppCtx를 주면 에러가 납니다.
              // 왜냐하면 myAppCtx은 MyApp위젯의 컨텍스트인데
              // 프로바이더는 MyApp과 Builder 사이에 있기 때문입니다.
              // 따라서 Builder 위젯의 컨텍스트를 전달해야 해당 프로바이더를 참조할 수 있습니다.
​
            return Scaffold(
              appBar: AppBar(
                title: Text('프로바이더 샘플'),
              ),
              body: Center(
                child: Consumer<CountProvider>(
                    //컨슈머 위젯을 사용하면 
                    // builder콜백이 리턴하는 위젯만이
                    // 구독하는 프로바이더에 의해 리빌드 됩니다.
                  builder: (consumerCtx, provider, child) {
                      // builder 콜백의 첫번째 인자는 Consumer위젯의 컨텍스트이고,
                      // 두번째 위젯은 구독하는 프로바이더를 참조합니다.
                      // 3번째 위젯은 추가적인 로직을 주고 싶을 때 사용합니다.
                      // child를 통해 조건문 등을 구현할 수 있겠죠?
                    return Text(
                      provider.count.toString(),
                      style: TextStyle(fontSize: 80),
                    );
                  },
                ),
                // child: Text(
                //   _countProvider.count.toString(),
                //   style: TextStyle(fontSize: 80),

              ),

                    icon: Icon(Icons.add),

                    },
                  ),
                  IconButton(
                    icon: Icon(Icons.remove),
                    onPressed: () {
                      _countProvider.remove();

              ),
            );

        ),
      ),
    );
